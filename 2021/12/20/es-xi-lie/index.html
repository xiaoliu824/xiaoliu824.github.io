ES6系列

一、let

1.用途

声明变量

2. 特点

①不能重复声明同一变量

let name = '张三';
//  let name = '法外狂徒'; 重复声明同一个变量会报错

②不存在变量提升
console.log(age); //报错
let age = 66;

③存在块级作用域
变量只在代码块中有效
在if else while for这些代码块里,用let声明的变量也存在块级作用域
{
   let boy = 'kobe'
   console.log(boy) //kobe
}
//  console.log(boy)  存在块级作用域,外面访问不到里面声明的变量

④不影响作用域链
{
  let school = '辉煌';
  function fn() {
    console.log(school)
  }
  fn() //辉煌
}

二、const

1. 用途

定义常量

2. 特点

①一定要赋初始值

// const A; //不赋初始值会报错

②常量要大写(潜规则),也可以小写
const Work = 'bug';
const hobby = '写bug';

③常量的值不能修改
const paly = '敲bug的工程师';
// paly ='头秃的bug工程师'; //报错

④存在块级作用域
{
  const lead = 'HONG';
  console.log(lead) //HONG
}
  //console.log(lead) //报错

⑤对于数组和对象的元素修改,不算做对常量的修改，不会报错
(因为它保存的值是一个地址值,指向并没有修改)
let Arr = ['科比','乔丹','卡特','麦迪'];
Arr.push('奥尼尔');

 //对数组/对象的元素修改,不算对常量修改,因为它保存的值是一个地址值,指向没有修改
 console.log(Arr);  //['科比', '乔丹', '卡特', '麦迪', '奥尼尔']

 const Team = ['科比','乔丹','卡特','麦迪'];
 // Team = 1000; //这样会报错,修改了其指向,

1. 什么是解构赋值

ES6允许按照一定模式从数组和对象中提取值,对变量进行赋值,这被称为解构赋值

2. 数组的解构

const F4 = ['小沈阳','刘能','赵本山','宋小宝'];

 // console.log(xiao); //小沈阳
 // console.log(liu); //刘能
 // console.log(zhao); //赵本山
 // console.log(song); //宋小宝

3. 对象的解构

const zxl = {
  name: '张小六',
  age: '不详',
  sayHello: function() {
   console.log('我可以say');
  }
 }

 let {name,age,sayHello} = zxl;
 console.log(name); //张小六
 console.log(age);  //不详
 console.log(sayHello);
 sayHello(); //我可以say

ES6中引入新的声明字符串的方式``

六、箭头函数

1. 箭头函数

①ES6允许使用箭头(=>)来声明函数

②箭头函数适合与this无关的回调,定时器、数组的方法回调

③箭头函数不适合与this有关的回调,例如:DOM事件回调、对象的方法

//声明一个函数
let fn = (a,b) => {
  return a+b;
}
//调用函数
let result = fn(3,2);
console.log(result); //5

2. 特点

Ⅰ. this是静态的,this始终指向函数声明时所在作用域下的this


var name = '张小6';
function fn1() {
  console.log(this.name); //张小6 this指向window
}
fn1();

let fn2 = () => {
  console.log(this.name); //张小6 this指向函数声明时的this,也就是window
}
fn2();

const school= {
  name: '张三',
}
fn1.call(school); //张三 将this的值改变为了school
fn2.call(school); //张小6 this始终指向函数声明时所在作用域下的this值

Ⅱ. 不能作为构造函数实例化对象
let Person = (name,age) => {
  this.name = name;
  this.age = age;
}
// let p1= new Person('6',33) //Person is not a constructor

Ⅲ. 不能使用arguments(保存实参)变量
let sum = () => {
  console.log(arguments);
} 
// sum(1,2,3); //arguments is not defined

3. 简写

3.1 省略小括号

当形参有且只有一个的时候,可以省略小括号

let number = n => {
  return n+1;
}
console.log(number(2)) //3

3.2 省略花括号

当代码体只有一条语句的时候,此时return必须省略,而且语句的执行结果就是函数的返回值

 let pow = n => n*n;
 console.log(pow(8)) //64
 let obj = {
  name: '6',
  getName:()=> {
    console.log(this); //此时this指向window
  }
}
console.log(obj.getName())

八、rest参数

ES6引入rest参数,用于获取函数的实参,用来代替 arguments

//ES5获取实参方式
 function getData() {
   console.log(arguments);
 }

 getData('小沈阳','赵本山');

 //rest 参数
 function Data(...args) {
   console.log(args);
 }

 Data(1,2,3,4,5,6)

 //rest 参数必须要放到参数最后
 function fn(a,b,...args) {
   console.log(a);
   console.log(b);
   console.log(args);
 }
 fn(1,2,3,4,5,6)

九、扩展运算符

1. 作用

... 扩展运算符能将数组转换为逗号分隔的参数序列

const szh = ['E','G','M',{name:'张四'}];

syc[3].name ='66';

③ 将伪数组转为真正的数组

const divs = document.querySelectorAll('div');
console.log(divs)
const divArr = [...divs];
console.log(divArr)

十、Symbol

1. 什么是Symbol

Symbol 表示独一无二的值,类似于字符串的数据类型

2. 特点

① Symbol的值是唯一的,用来解决命名冲突的问题

② Symbol的值不能与其他数据进行运算

③ Symbol定义的对象属性不能使用for…in循环遍历,但是可以使用Reflect.ownKeys来获取对象的所有键名

let s = Symbol();
console.log(s, typeof s); //Symbol() 'symbol'
let s1 = Symbol('测试');
console.log(s1); //Symbol(测试)
//Symbol.for 创建
let s2 = Symbol.for('测试001');
console.log(s2) //Symbol(测试001)
//不能和其他数据类型进行运算
// let result = s + 100;
// let result = s > 10;
// let result = s + s;
// console.log(result); //报错

④ 向对象中添加方法

//向对象中添加方法 up down
  let game = {
    name: '俄罗斯方块',
    up:function() {
      console.log('上');
    },
    down: ()=> console.log('下')
  }
  let methods = {
    up: Symbol(),
    down: Symbol(),
  }
  game[methods.up] = () => console.log('我可以改变形状')
  game[methods.down] = () => console.log('我可以快速向上')
  console.log(game)

⑤ 创建对象属性
//Symbol创建对象属性
 let youxi = {
   name: '狼人杀',
   //[Symbol('say')] Sybol是一个动态的,通过这种方式添加属性
   [Symbol('say')]: function() {
     console.log('我可以发言');
   },
   [Symbol('zibao')]: function() {
     console.log('我可以自爆');
   }
 }
 console.log(youxi)

十一、Iterator

1. 什么是Iterator

iterator接口在JS中相当于对象的一个属性,属性名是Symbol.iterator,属性值是一个函数

let xyj = ['唐僧','孙悟空','猪八戒','沙僧'];
for(let v in xyj) { //for...in循环前面的变量v保存的是,键名
  console.log(v); //0 1 2 3
}
for(let v of xyj) { //for...of循环前面的变量v保存的是,键值
  console.log(v); //'唐僧','孙悟空','猪八戒','沙僧'
}

2. 工作原理(迭代器遍历数据的原理)

①创建一个指针对象,指向当前数据结构的起始位置

②第一次调用它的next方法,指针指向数据结构的第一个成员

③不断调用next方法,指针一直往后移动,直到最后一个成员

④每调用一个next方法,返回一个包含value和done属性的值

console.log(iterator.next()); //{value: '唐僧', done: false}
console.log(iterator.next()); //{value: '唐僧', done: false}
console.log(iterator.next()); //{value: '唐僧', done: false}
console.log(iterator.next()); //{value: undefined, done: true}
console.log(iterator)

十二、生成器

1. 生成器

生成器也是一个特殊的函数,这个函数主要用来做异步编程,函数代码的分隔符yield,它返回的结果是一个迭代器对象

function * gen() {
    yield '一只没有耳朵';
    yield '一只没有尾巴';
    yield '真奇怪';
  }
  let iterator = gen(); //它返回的结果是一个迭代器对象

  //调用next()方法才会运行
  console.log(iterator.next()); {value: '一只没有耳朵', done: false}

  console.log(iterator.next()); {value: '一只没有尾巴', done: false}

  console.log(iterator.next()); {value: '真奇怪', done: false}

  console.log(iterator.next()); {value: undefined, done: true}

2. 生成器函数参数

next可以传入参数,传入参数将作为上一个yield语句的返回结果

//生成器函数参数
 function * Add(args) {
   console.log(args); //AAA
   let one =yield 111;
   console.log(one) //BB
   let two= yield 222;
   console.log(two); //CC
   let three= yield 333;
   console.log(three) //DD
 }

 //获取迭代器对象
 let result = Add('AAA');

 console.log(result.next()) //{value: 111, done: false}
 
 //next也可以传入参数,传入参数将作为上一个yield语句的返回结果
 console.log(result.next('BB')) //{value: 222, done: false}

 console.log(result.next('CC')) //{value: 333, done: false}

 console.log(result.next('DD')) //{value: undefined, done: true}

十三、Promise基本语法

1. 实例化Promise对象

new Promise(参数) 参数为一个函数,函数有两个形参resolve,reject 这两个参数也是函数

2. 对象状态

初始化(pending)、成功(fulfilled)、失败(rejected)

3. 状态变更

调用resolve、reject这两个函数,来改变promise对象的状态

①调用resolve，promise对象的状态就会变为成功,成功后调用promise对象的then方法,之后会执行then方法中第一个回调函数

②调用reject，promise对象的状态就会变为失败,失败后调用promise对象的then方法,之后会执行then方法中第二个回调函数

4. then方法

  1. promise对象的then方法,then方法接收两个参数,两个参数都是函数类型的值

    成功的形参一般是value,失败的形参一般是reason

    const p = new Promise(function(resolve,reject){
    setTimeout(() => {
      //成功状态
      //  let data = '数据拉取成功';
      //  resolve(data); //将对象的状态变为成功,之后执行then方法中的第一个回调函数
      //失败状态
       let data2 = '数据拉取失败';
       reject(data2);
    }, 1000);
  })

  p.then(function(value){
    console.log(value);
  },function(reason){
    console.log(reason)
  })

  2. 调用then方法,then方法返回的结果是promise对象,对象的状态由回调函数的执行结果决定

    ① 如果回调函数中返回的结果是非promise类型数据,状态为成功,返回值为对象的成功值

    ② 如果回调函数种返回的结果是promise对象,内部返回的promise状态决定then方法返回promise的状态

    ③ 如果回调函数内部返回的结果是抛出错误,状态也是失败,返回值为抛出的值

    //可以通过链式调用解决回调地狱
 let result=p.then((value)=> {
   console.log(value)
   // return new Promise((resolve,reject) => {
   //   reject('error')//二内部promise
   // })
   throw new Error('出错了'); //三抛出异常
 },(reason)=> {
   console.warn(reason)
 })

 //第一种结果
 //返回状态为成功,因为未设置返回值,函数默认返回undefined，它为非promise类型数据
 //所以返回值为undefined
 console.log(result);

    5. catch方法

    const p =new Promise((resolve,reject) => {
     const data = '失误拉';
     reject(data);
 })
 p.catch(error=> {
  console.log(error)
 })

    十六、class 类

    1. 什么是class

    ES6引入了Class(类)这个概念,作为对象的模板.
    通过class关键字,可以定义类,ES6的class可以看作只是一个语法糖,它的绝大部分功能,ES5都可以做到,新的class写法让对象原型写法更加清晰。

2. constructor 定义构造函数初始化

① ES5实现

function Phone(band,price) {
     this.band = band;
     this.price = price;
   }
   //添加方法
   Phone.prototype.call = function() {
     console.log('我可以打电话');
   }
   //创建实例
   let ip12 = new Phone('苹果12','6999');
   ip12.call()
   console.log(ip12)

② ES6实现

//class
class ChPhone{//ChPhone 类名
  //构造方法 名字不能修改 (该方法自动执行的,当使用new 类名时执行)
  constructor(bind,price) {
    this.bind = bind;
    this.price = price;
  }
  //方法必须使用该语法,不能使用ES5的对象完整形式 call:function() {} 
  //class中不支持Es5这种完整形式
  call() {
    console.log('我可以打电话啊!!!')
  }
}
const hw = new ChPhone('华为5g','6999');
hw.call()
console.log(hw);

3. extends 继承父类

① ES5构造函数实现继承

//定义父类
   function PHone(brand,price) {
     this.brand = brand;
     this.price = price;
   }
   PHone.prototype.call = function() {
     console.log('我可以打电话');
   }
   //定义子类
   function smallPhone(brand,price,color,size) {
     PHone.call(this,brand,price) //这里的this指向smallPhone的实例对象
     this.color = color;
     this.size = size;
   }
   //设置子类的原型对象为父类的实例对象
   smallPhone.prototype = new PHone; 
   smallPhone.prototype.constructor = smallPhone;

   smallPhone.prototype.photo = function(){
     console.log('我可以拍照');
   }
   let chuizi = new smallPhone('锤子',3033,'紫色','6.6')
   console.log(chuizi)

② ES6实现

//定义父类
  class Phone{
    //构造方法
    constructor(brind,price) {
      this.brind = brind;
      this.price = price;
    }
    //父类成员的属性
    call() {
      console.log('我可以打电话')
    }
  }
  //定义子类
  class SmallPhone extends Phone{ //extends继承
    //构造方法
    constructor(brand,price,color,size) {
      //super就是父类的constructor方法(构造方法)
      super(brand,price) //相当于Phone.call(this,brand,price)
      this.color = color;
      this.size = size;
    }
    photo() {
      console.log('我可以拍照')
    }
    playG() {
      console.log('我可以玩游戏')
    }
    //对父的同名方法进行重写
    //在js的class语法中,子类不可以调用父类的同名方法 
    //在普通的成员方法中不允许出现super去调用父类的同名方法
    call() {
      console.log('我可以视频')
    }
  }

4. super 调用父级构造方法

//定义父类
  class Phone{

ES9 允许命名捕获组使用符号『?<name>』,这样获取捕获结果可读性更强

//声明一个字符串
  let str = '<a href="https://xiaoliu824.github.io/">张三</a>'
  //提取 url与 标签文本
  //(.*) 在写正则同时还可以对某些数据进行单独读取捕获
  const reg = /<a href="(.*)">(.*)<\/a>/;
    //exec方法返回一个数组
  const result = reg.exec(str); //0为匹配的字符串 1为url 2为标签文本

  let str2 = '<a href="https://xiaoliu824.github.io/">张三</a>'
  const reg2 = /<a href="(?<url>.*)">(?<text>.*)<\/a>/;
  let res = reg2.exec(str2)
  console.log(res)

  //第一个和第二个的区别 第一个的groups为undefined 第二个groups有值

2. 正则表达式反向断言

ES9 支持反向断言，通过对匹配结果前面的内容进行判断，对匹配进行筛选

//断言
//可以根据目标内容的前边和后边来对它做一个唯一性的识别

//声明字符串
let str = 'js5211314你知道吗555啦啦啦';
//正向断言(根据当前匹配后边的内容来去判断前边内容是不是合法的,是不是满足条件的)
let reg = /\d+(?=啦)/  //判断后面的字是不是啦,如果是啦的话才满足条件
let result = reg.exec(str);
console.log(result)
//['555', index: 13, input: 'js5211314你知道吗555啦啦啦', groups: undefined]

//反向断言(根据当前匹配前面的内容来判断后边内容是不是合法的,是不是满足条件的)

//(?<=前边内容) 固定语法

let res = /(?<=吗)\d+/
let rest = res.exec(str);
console.log(rest)
//['555', index: 13, input: 'js5211314你知道吗555啦啦啦', groups: undefined]

    console.log(res)

console.log(arr.flat()) // [1, 2, 3, 4, 5, 6, 7, 8, 9]
let arr1 = [1,2,4,6,9,['z','x',[1,'q','w']]];
//将三维数组转换成一维数组
console.log(arr1.flat(2)); //[1, 2, 4, 6, 9, 'z', 'x', 1, 'q', 'w']

const sz= [1,2,3,4,5]
//map方法返回一个新数组,如果map返回的是一个多维数组,可以用flatMap将它变成一个一维的
const res = sz.flatMap(item => [item * 5])

2. 类的私有属性

私有属性,对属性的一种封装,不被外部直接操作

//私有属性可以理解为
  //一个女孩子,它是一个对象(语言上的对象),她有一些属性就是不能直接被别人访问的(体重、年龄)
  class Person {
    //公有属性
    name;
    //私有属性 前面必须加#
    #age;
    #weight;
    //构造方法
    constructor(name,age,weight) {
      this.name = name;
      this.#age = age;
      this.#weight = weight;
    }
    info() {
      console.log(this.name);
      console.log(this.#age); //在类内部可以访问到,类外部不可以访问
      console.log(this.#weight)
    }
  }

  const girl = new Person('小红',18,'45kg')
  console.log(girl) //{name: '小红', #age: 18, #weight: '45kg'}
  console.log(girl.name); //小红
  // console.log(girl.#age); //在类外部不能访问,在类里面可以访问
  girl.info()

3. Promise.allSettled

//Promise.allSettled 该方法接收一个Promise的数组
//返回的结果也是一个Promise对象,
//不过,返回的结果(不管中间有没有失败的promise)永远都是成功的状态,

//成功的值是数组中每个promise的结果状态(成功和失败)和结果值
let p =new Promise((resolve,reject) => {
  setTimeout(() => {
    resolve('商品数据1')
  }, 1000);
})
let p2 = new Promise((resolve,reject) => {
  setTimeout(() => {
    resolve('用户数据1')
  }, 1000);
})

//返回结果始终是成功的,成功的值是每一个promise对象执行的结果(异步任务执行完的值和执行状态)
let result = Promise.allSettled([p,p2])

//Promise.all  接收一个Promise数组
//与allSettled的区别是?
//它的返回结果根据每一个promise对象状态决定的,如果每一个promise都是成功的状态,返回的结果也是成功的promise对象
//如果有一个是失败的,返回的promise就是失败的
//全部成功才会返回成功的promise
console.log(result)

6. BigInt

BigInt 大整数,大整形 主要用于大数值运算

//表示方式 在普通整型数组的基础上后面加n,n表示大整形
 let num = 123n;
 console.log(num,typeof num) //123n 'bigint'

 //BigInt() 函数  可以将一个普通整数值转换为大整形的值
 //不能使用浮点数(小数)来转换
 console.log(BigInt(125)); //125n
 // console.log(BigInt(1.4)) //报错

 //Number.MAX_SAFE_INTEGER 最大安全整数
 const max = Number.MAX_SAFE_INTEGER;
 console.log(max) //9007199254740991
 console.log(max+1)
 console.log(max+2) //运算出错,不能再表示更大运算正确的运算结果

 //可以借助BigInt来运算更大的结果
 //BigInt类型不能直接和普通Int类型做运算,必须将Int类型转换为BigInt类型
 console.log(BigInt(max)) //9007199254740991n
 console.log(BigInt(max)+BigInt(1)); //9007199254740992n
 console.log(BigInt(max)+BigInt(2)); //9007199254740993n
