ES6系列

一、let

1.用途

声明变量

2. 特点

①不能重复声明同一变量

let name = '张三';
//  let name = '法外狂徒'; 重复声明同一个变量会报错

②不存在变量提升
console.log(age); //报错
let age = 66;

③存在块级作用域
变量只在代码块中有效
在if else while for这些代码块里,用let声明的变量也存在块级作用域
{
   let boy = 'kobe'
   console.log(boy) //kobe
}
//  console.log(boy)  存在块级作用域,外面访问不到里面声明的变量

④不影响作用域链
{
  let school = '武汉辉煌';
  function fn() {
    console.log(school)
  }
  fn() //武汉辉煌
}

二、const

1. 用途

定义常量

2. 特点

①一定要赋初始值

// const A; //不赋初始值会报错

②常量要大写(潜规则),也可以小写
const Work = 'bug';
const hobby = '写bug';

③常量的值不能修改
const paly = '敲bug的工程师';
// paly ='头秃的bug工程师'; //报错

④存在块级作用域
{
  const lead = 'HONG';
  console.log(lead) //HONG
}
  //console.log(lead) //报错

⑤对于数组和对象的元素修改,不算做对常量的修改，不会报错
(因为它保存的值是一个地址值,指向并没有修改)
let Arr = ['科比','乔丹','卡特','麦迪'];
Arr.push('奥尼尔');

 //对数组/对象的元素修改,不算对常量修改,因为它保存的值是一个地址值,指向没有修改
 console.log(Arr);  //['科比', '乔丹', '卡特', '麦迪', '奥尼尔']

 const Team = ['科比','乔丹','卡特','麦迪'];
 // Team = 1000; //这样会报错,修改了其指向,

三、解构赋值

1. 什么是解构赋值

ES6允许按照一定模式从数组和对象中提取值,对变量进行赋值,这被称为解构赋值

2. 数组的解构

const F4 = ['小沈阳','刘能','赵本山','宋小宝'];
 // let [xiao,liu,zhao,song] = F4;
 // console.log(xiao); //小沈阳
 // console.log(liu); //刘能
 // console.log(zhao); //赵本山
 // console.log(song); //宋小宝

3. 对象的解构

const zxl = {
  name: '张小六',
  age: '不详',
  sayHello: function() {
   console.log('我可以say');
  }
 }

 let {name,age,sayHello} = zxl;
 console.log(name); //张小六
 console.log(age);  //不详
 console.log(sayHello);
 sayHello(); //我可以say

四、模板字符串

1. 声明

ES6中引入新的声明字符串的方式``

2. 特性

①内容中可以直接出现换行符(可以进行换行)

let content = `<ul>
    <li>沈腾</li>
    <li>玛丽</li>
    </ul>`

②可以进行变量拼接,使用${变量}
let player= 'kobe';
let say = `${player}是我心中伟大的球星`;
console.log(say) //kobe是我心中伟大的球星

五、简化对象写法

1.ES6允许在大括号里面,直接写入变量和函数,作为对象的属性和方法

2.当属性名和变量名相同时,可以简写,原来写法 name: name

let name ="张小6";
let change = function() {
  console.log('我们可以改变你');
}
let peo = {
   name, //当属性名和变量名相同时,可以简写,原来写法name: name,
   change,
}
console.log(peo)

六、箭头函数

1. 箭头函数

①ES6允许使用箭头(=>)来声明函数

②箭头函数适合与this无关的回调,定时器、数组的方法回调

③箭头函数不适合与this有关的回调,例如:DOM事件回调、对象的方法

//声明一个函数
let fn = (a,b) => {
  return a+b;
}
//调用函数
let result = fn(3,2);
console.log(result); //5

2. 特点

Ⅰ. this是静态的,this始终指向函数声明时所在作用域下的this

var name = '张小6';
function fn1() {
  console.log(this.name); //张小6 this指向window
}
fn1();

let fn2 = () => {
  console.log(this.name); //张小6 this指向函数声明时的this,也就是window
}
fn2();

const school= {
  name: '张三',
}
fn1.call(school); //张三 将this的值改变为了school
fn2.call(school); //张小6 this始终指向函数声明时所在作用域下的this值

Ⅱ. 不能作为构造函数实例化对象
let Person = (name,age) => {
  this.name = name;
  this.age = age;
}
// let p1= new Person('6',33) //Person is not a constructor

Ⅲ. 不能使用arguments(保存实参)变量
let sum = () => {
  console.log(arguments);
} 
// sum(1,2,3); //arguments is not defined

3. 简写

3.1 省略小括号

当形参有且只有一个的时候,可以省略小括号

let number = n => {
  return n+1;
}
console.log(number(2)) //3

3.2 省略花括号

当代码体只有一条语句的时候,此时return必须省略,而且语句的执行结果就是函数的返回值

 let pow = n => n*n;
 console.log(pow(8)) //64
 let obj = {
  name: '6',
  getName:()=> {
    console.log(this); //此时this指向window
  }
}
console.log(obj.getName())

七、参数默认值

ES6允许给函数参数赋初始值(如果传了值,就用传过来的值,没有的话就用初始值)

①形参初始值,具有默认值的参数一般要靠后放(潜规则)

function add(a,b,c=10) {
  return a+b+c;
}
let result = add(1,5);
console.log(result) //16

②与解构赋值结合
function content({name,age,hoby,dream = "月入w"}) {
  console.log(name); //小6
  console.log(age);  //不祥
  console.log(hoby); //写bug
  console.log(dream); //月入w
}
content({
 name: '小6',
 age: '不祥',
 hoby: '写bug',
})

八、rest参数

ES6引入rest参数,用于获取函数的实参,用来代替 arguments

//ES5获取实参方式
 function getData() {
   console.log(arguments);
 }

 getData('小沈阳','赵本山');

 //rest 参数
 function Data(...args) {
   console.log(args);
 }

 Data(1,2,3,4,5,6)

 //rest 参数必须要放到参数最后
 function fn(a,b,...args) {
   console.log(a);
   console.log(b);
   console.log(args);
 }
 fn(1,2,3,4,5,6)

九、扩展运算符

1. 作用

```
 //声明一个数组
 const F4 = ['小沈阳','赵本山','刘能','赵四'];
 //'小沈阳','赵本山','刘能','赵四'
 function Chunw() {
   console.log(arguments);
 }
 Chunw(...F4) //'小沈阳','赵本山','刘能','赵四'

2. 扩展运算符的运用

① 数组的合并

const kuaizi = ['肖央','王太利'];
const F2 = ['小6','张三'];
const Arr = [...kuaizi,...F2];
console.log(Arr)//['肖央', '王太利', '小6', '张三']

② 数组的克隆

const szh = ['E','G','M',{name:'张四'}];
const syc = [...szh]; //如果数组中含有引用数据类型的 也是属于浅拷贝
syc[3].name ='66';
console.log(szh)
console.log(syc)

③ 将伪数组转为真正的数组

const divs = document.querySelectorAll('div');
console.log(divs)
const divArr = [...divs];
console.log(divArr)

十、Symbol

1. 什么是Symbol


### 2. 特点
 ① Symbol的值是唯一的,用来解决命名冲突的问题

 ② Symbol的值不能与其他数据进行运算

 ③ Symbol定义的对象属性不能使用for...in循环遍历,但是可以使用Reflect.ownKeys来获取对象的所有键名

let s = Symbol();
console.log(s, typeof s); //Symbol() ‘symbol’
let s1 = Symbol(‘测试’);
console.log(s1); //Symbol(测试)
//Symbol.for 创建
let s2 = Symbol.for(‘测试001’);
console.log(s2) //Symbol(测试001)
//不能和其他数据类型进行运算
// let result = s + 100;
// let result = s > 10;
// let result = s + s;
// console.log(result); //报错


④ 向对象中添加方法

//向对象中添加方法 up down
let game = {
name: ‘俄罗斯方块’,
up:function() {
console.log(‘上’);
},
down: ()=> console.log(‘下’)
}
let methods = {
up: Symbol(),
down: Symbol(),
}
game[methods.up] = () => console.log(‘我可以改变形状’)
game[methods.down] = () => console.log(‘我可以快速向上’)
console.log(game)

⑤ 创建对象属性

//Symbol创建对象属性
let youxi = {
name: ‘狼人杀’,
//[Symbol(‘say’)] Sybol是一个动态的,通过这种方式添加属性

console.log(‘我可以发言’);
},
[Symbol(‘zibao’)]: function() {
console.log(‘我可以自爆’);
}
}

## 十一、Iterator
### 1. 什么是Iterator
 iterator接口在JS中相当于对象的一个属性,属性名是Symbol.iterator,属性值是一个函数

let xyj = [‘唐僧’,’孙悟空’,’猪八戒’,’沙僧’];
for(let v in xyj) { //for…in循环前面的变量v保存的是,键名
console.log(v); //0 1 2 3
}
for(let v of xyj) { //for…of循环前面的变量v保存的是,键值
console.log(v); //‘唐僧’,’孙悟空’,’猪八戒’,’沙僧’
}

### 2. 工作原理(迭代器遍历数据的原理)
 ①创建一个指针对象,指向当前数据结构的起始位置

 ②第一次调用它的next方法,指针指向数据结构的第一个成员

 ③不断调用next方法,指针一直往后移动,直到最后一个成员

 ④每调用一个next方法,返回一个包含value和done属性的值

console.log(iterator.next()); //{value: ‘唐僧’, done: false}
console.log(iterator.next()); //{value: ‘唐僧’, done: false}
console.log(iterator.next()); //{value: ‘唐僧’, done: false}
console.log(iterator.next()); //{value: undefined, done: true}
console.log(iterator)

## 十二、生成器
### 1. 生成器
 生成器也是一个特殊的函数,这个函数主要用来做异步编程,函数代码的分隔符yield,它返回的结果是一个迭代器对象

function * gen() {
yield ‘一只没有耳朵’;
yield ‘一只没有尾巴’;
yield ‘真奇怪’;
}
let iterator = gen(); //它返回的结果是一个迭代器对象

//调用next()方法才会运行
console.log(iterator.next()); {value: '一只没有耳朵', done: false}

console.log(iterator.next()); {value: '一只没有尾巴', done: false}

console.log(iterator.next()); {value: '真奇怪', done: false}

console.log(iterator.next()); {value: undefined, done: true}

### 2. 生成器函数参数
 next可以传入参数,传入参数将作为上一个yield语句的返回结果

//生成器函数参数
function * Add(args) {
console.log(args); //AAA
let one =yield 111;
console.log(one) //BB
let two= yield 222;
console.log(two); //CC
let three= yield 333;
console.log(three) //DD
}

//获取迭代器对象
let result = Add('AAA');

console.log(result.next()) //{value: 111, done: false}

//next也可以传入参数,传入参数将作为上一个yield语句的返回结果
console.log(result.next('BB')) //{value: 222, done: false}

console.log(result.next('CC')) //{value: 333, done: false}

console.log(result.next('DD')) //{value: undefined, done: true}

## 十三、Promise基本语法
### 1. 实例化Promise对象
 new Promise(参数) 参数为一个函数,函数有两个形参resolve,reject 这两个参数也是函数

### 2. 对象状态
 初始化(pending)、成功(fulfilled)、失败(rejected)

### 3. 状态变更
 调用resolve、reject这两个函数,来改变promise对象的状态

  ①调用resolve，promise对象的状态就会变为成功,成功后调用promise对象的then方法,之后会执行then方法中第一个回调函数

  ②调用reject，promise对象的状态就会变为失败,失败后调用promise对象的then方法,之后会执行then方法中第二个回调函数

### 4. then方法
 1. promise对象的then方法,then方法接收两个参数,两个参数都是函数类型的值

 成功的形参一般是value,失败的形参一般是reason

const p = new Promise(function(resolve,reject){
setTimeout(() => {
//成功状态
// let data = ‘数据拉取成功’;
// resolve(data); //将对象的状态变为成功,之后执行then方法中的第一个回调函数
//失败状态
let data2 = ‘数据拉取失败’;
reject(data2);
}, 1000);
})

p.then(function(value){
  console.log(value);
},function(reason){
  console.log(reason)
})

2. 调用then方法,then方法返回的结果是promise对象,对象的状态由回调函数的执行结果决定

  ① 如果回调函数中返回的结果是非promise类型数据,状态为成功,返回值为对象的成功值

  ② 如果回调函数种返回的结果是promise对象,内部返回的promise状态决定then方法返回promise的状态

  ③ 如果回调函数内部返回的结果是抛出错误,状态也是失败,返回值为抛出的值

//可以通过链式调用解决回调地狱
let result=p.then((value)=> {
console.log(value)
// return new Promise((resolve,reject) => {
// reject(‘error’)//二内部promise
// })
throw new Error(‘出错了’); //三抛出异常
},(reason)=> {
console.warn(reason)
})

//第一种结果
//返回状态为成功,因为未设置返回值,函数默认返回undefined，它为非promise类型数据
//所以返回值为undefined
console.log(result); 

### 5. catch方法

const p =new Promise((resolve,reject) => {
const data = ‘失误拉’;
reject(data);
})
p.catch(error=> {
console.log(error)
})


## 十四、Set
### 1. 什么是Set
 ES6提供了新的数据结构Set(集合),它类似于数组，但成员的值都是唯一的,集合实现了iterator接口,可以使用```扩展运算符```和```for···of```进行遍历

//声明一个Set
let s = new Set();
let s2 = new Set([‘好事儿’,’坏事儿’,’小事儿’,’大事儿’,’坏事儿’]);
//Set会进行自动去重
console.log(s2) //{‘好事儿’, ‘坏事儿’, ‘小事儿’, ‘大事儿’}

### 2. 集合的属性和方法
#### ① size
 返回集合的元素个数

//元素的个数
console.log(s2.size)

#### ② add
 增加一个新元素,返回当前集合

//添加新的元素
s2.add(‘烦心事’);
console.log(s2) //{‘好事儿’, ‘坏事儿’, ‘小事儿’, ‘大事儿’, ‘烦心事’}

#### ③ delete
 删除元素,返回boolean值

//删除元素
s2.delete(‘坏事儿’);
console.log(s2); //{‘好事儿’, ‘小事儿’, ‘大事儿’, ‘烦心事’}

#### ④ has
 检测集合中是否包含某个元素,返回boolean值

//检测集合中是否有某个元素，返回布尔值，有则返回true,没有则返回false
let result =s2.has(‘小事儿’);
console.log(result)

#### ⑤ clear
 清空集合,返回undefined

result = s2.clear();
console.log(result) //undefined

## 十五、Map
### 1.什么是Map
 ES6提供了Map数据结构,它类似于对象,也是键值对集合.但是"键"的范围不限于字符串,各种类型的值(包括对象)都可以当作键.Map也实现了iterator接口,可以使用```扩展运算符```和```for···of```进行遍历

//声明map
let m = new Map();

### 2. Map属性和方法
#### ① size
 返回Map的元素个数

//size
console.log(m.size)

#### ② set
 增加一个新元素,返回当前Map

m.set(‘name’,’张小6’); //key: “name” value: “张小6”
m.set(‘change’,function() {
console.log(‘改变’);
}) //{“change” => function() { console.log(‘改变’); }}
let school = {
name: ‘NBA’
}
m.set(school,[‘Kobe’,’Jordan’,’James’])
//key: {name: ‘NBA’} value: (3) [‘Kobe’, ‘Jordan’, ‘James’]

#### ③ get
 返回键名对应的键值

//获取
console.log(m.get(school)) //[‘Kobe’, ‘Jordan’, ‘James’]

#### ④ has
 检测Map中是否包含某个元素,返回boolean值

//检测
let result = m.has(school)
console.log(result) //true

#### ⑤ clear
 清空集合,返回undefined

//清空
m.clear();
