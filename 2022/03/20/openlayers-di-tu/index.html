① 安装插件

npm install ol -S

② 显示地图

<template>
<!--第一种显示地图方式-->
  <div class="main_box">
    <div ref="map" id="map"></div>
    <div ref="popup" class="popup" v-show="showPopup">
      <div class="info"></div>
    </div>
  </div>
</template>

<script>
import "ol/ol.css";
import OSM from "ol/source/OSM";
import TileLayer from "ol/layer/Tile";
import VectorLayer from "ol/layer/Vector";
import VectorSource from "ol/source/Vector";
import { Map, View, Feature } from "ol";
import { Style, Icon } from "ol/style";
import { Point } from "ol/geom";
import Overlay from "ol/Overlay";
import { Fill, Stroke, Circle as sCircle } from "ol/style";

export default {
  name: 'maplayers',
  data() {
    return {
      //* 第一种
      map: null,
      // url: '/static/openStreetMapsichuan/{z}/{x}/{y}.png', // 加载离线瓦片地图
      url: "https://webrd01.is.autonavi.com/appmaptile?lang=zh_cn&size=1&scale=1&style=8&x={x}&y={y}&z={z}", // 加载在线瓦片地图
      // 地图View展示选项配置
      viewOptions: {
        projection: "EPSG:4326",
        // center: [102.87855327334883, 29.95539168988012], // [104.089175, 30.650451]
        // center: [104.089175, 30.650451], // [104.089175, 30.650451] 地图中心位置 成都
        center: [114.21, 30.37], // 武汉
        zoom: 10,
        // extent: [101.9004807124176, 28.837774208959473, 103.38848561448468, 30.9210929107385], // [minX, minY, maxX, maxY]
        // minZoom: 6,
        // maxZoom: 19 // 缩放最大级别控制
      },
      view: null,
      source: null,
      pielayer: null, // 点线信息的图层,
      pointLayer: {},
      overlay: null, // 弹窗
      markerSource: null, // 坐标数据源
      showPopup: false,
      position: null,
      popup: null,
      //* 第二种
      // openMap: null
    };
  },
  methods: {
    // 绘制地图
    initMap() {
      //* 1
      let _this = this;
      var layers = new TileLayer({
        source: new OSM({
          url: _this.url,
        }),
      });
      _this.view = new View(_this.viewOptions);
      this.map = new Map({
        layers: [layers],
        target: "map",
        view: _this.view,
      });
      // this.setMarker()
      this.addOverlay();
      this.map.on("pointermove", this.mapMoveIn);

      //* 2
      // this.openMap = new Map({
      //   target: 'map',
      //   layers: [
      //     new TileLayer({
      //       source: new XYZ({
      //         url: 'https://wprd04.is.autonavi.com/appmaptile?lang=zh_cn&size=1&style=7&x={x}&y={y}&z={z}'
      //       })
      //     })
      //   ],
      //   view: new View({
      //     center:[108.945951,34.465262],
      //     zoom:1,
      //   }),
      //   controls: []
      // })
    },
    // 为地图添加点击事件
    clickMap() {
      this.map.on("click", (e) => {
        console.log(e.coordinate)
        // 经纬度
        this.addPoints(e.coordinate);
        this.isShow = true;
      });
    },
    // 根据经纬度坐标添加摄像头要素 || 地图标点
    addPoints(coordinate) {
      if (Object.keys(this.pointLayer).length == 0) {
        // 创建图层
        this.pointLayer = new VectorLayer({
          source: new VectorSource(),
        });
        // 图层添加到地图上
        this.map.addLayer(this.pointLayer);
      }
      // 创建feature要素,一个feature就是一个点坐标信息
      const feature = new Feature({
        geometry: new Point(coordinate),
      });
      // 设置要素的图标
      feature.setStyle(
        new Style({
          // // 设置图片效果
          // image: new Icon({
          //   // src: 'https://smart-garden-manage.oss-cn-chengdu.aliyuncs.com/shexiangtou.png',
          //   // src: 'https://img0.baidu.com/it/u=2244716523,839477405&fm=253&fmt=auto&app=138&f=JPEG?w=400&h=500',
          //   src: "https://openlayers.org/en/latest/examples/data/icon.png", // 地图标点
          //   scale: 0.5,
          // }),
          image: new sCircle({
            radius: 10,
            stroke: new Stroke({
              color: "#fff",
            }),
            fill: new Fill({
              color: "#3399CC",
            }),
          }),
        })
      );
      // 要素添加到地图图层上
      // this.pointLayer.getSource().addFeatures([feature]);
      this.pointLayer.getSource().addFeature(feature);
    },
    mapMoveIn(e) {
      const features = this.map.getFeaturesAtPixel(e.pixel);
      // 移到点位上才会触发
      if (
        this.map.hasFeatureAtPixel(e.pixel) &&
        features[0].getGeometry() instanceof Point
      ) {
        this.showPopup = true;
        let coordinates = features[0].getGeometry().getCoordinates();
        this.position = coordinates;
        this.popup.setPosition(coordinates);
      } else {
        this.showPopup = false;
      }
    },
    // 将鼠标样式进行更改
    pointrmove() {
      this.map.on("pointermove", (e) => {
        if (this.map.hasFeatureAtPixel(e.pixel)) {
          // 手
          this.map.getViewport().style.cursor = "pointer";
        } else {
          // 正常选择,箭头
          this.map.getViewport().style.cursor = "inherit";
        }
      });
    },

    addOverlay() {
      // 创建Overlay
      let elPopup = this.$refs.popup;
      this.popup = new Overlay({
        element: elPopup,
        positioning: "bottom-center",
        stopEvent: false,
        offset: [0, -20],
      });
      this.map.addOverlay(this.popup);
    },
  },
  mounted() {
    this.initMap();
    this.clickMap();
    this.pointrmove();
  },
};
</script>

 import {
  Feature
} from 'ol'
import * as olProj from 'ol/proj'
import {
  Point
} from 'ol/geom'
import VectorSource from 'ol/source/Vector'
import {
  Vector as VectorLayer
} from 'ol/layer'
import {
  Style,
  Circle as sCircle,
  Stroke,
  Fill,
  RegularShape
} from 'ol/style'

/**
 * 显示不同形态的数据
 */
export function showdifferent(Geodata) {
  if (Geodata.length) {
    let feature, Pointzb = [];
    // 所有坐标
    // let coordAll = collectcoord(Geodata)
    Geodata.forEach(itemgeo => {
      if (itemgeo.GeoType === 1) { // 点
        // Pointzb.push(itemgeo.Geometry)
        Pointzb.push({
          coord: itemgeo.Geometry,
          color: itemgeo.GeoColor,
          shapeType: itemgeo.GeoStype,
          size: itemgeo.GeoSize
        })
      } else if (itemgeo.GeoType === 2) { // 线

      } else if (itemgeo.GeoType === 3) { // 面

      }
    })
    feature = disPoint(Pointzb)
    // console.log(feature)
    // 矢量图层
    let _marker = new VectorLayer({
      // 矢量源
      source: new VectorSource({
        features: feature
      })
    })

    return _marker
  }
}


// 处理
export function disPoint(GeoAllcoord) {
  let _features = GeoAllcoord.reduce((feaList, item) => {
    // 实例化要素
    let _feature = new Feature({
      // fromLonLat 将坐标从经度/纬度转换为其他投影
      geometry: new Point(olProj.fromLonLat(item.coord))
    })
    // 设置样式
    let style = Settstyle(item.color,item.shapeType,item.size)
    console.log(style)
    _feature.setStyle(style)
    console.log(_feature)
    feaList.push(_feature)
    return feaList
  }, [])

  return _features
}

// 设置样式
export function Settstyle(colorGeo = '#3399CC', Type,Size = 8) {
  // 样式   lineColor-> 边的颜色  lineSize->边的尺寸
  let _style,lineColor = "#fff",lineSize = 3;
  let StyleConfig = [
    {id: '101',shape: 'RegularShape',point: 5,logo: 'star'},
    {id: '102',shape: 'sCircle'},
    {id: '103',shape: 'RegularShape',point: 4,logo: 'diamond'},
    {id: '104',shape: 'RegularShape',point: 3,logo: 'triangle'}
  ]
  for(let i =0; i< StyleConfig.length; i++) {
    let StyleObj = StyleConfig[i]
    if(StyleObj.id == Type) {
      if(StyleObj.shape == 'RegularShape') {
        if(StyleObj.logo == 'star') {
          _style = new Style({
            image: new RegularShape({ 
              points: StyleObj.point || 5, // 顶点数
              radius1: Size, // 外圈大小
              radius2: Size/2, // 内圈大小
              stroke: new Stroke({ // 设置边的样式
                color: lineColor,
                size: lineSize
              }),
              fill: new Fill({ // 设置填充颜色
                color: colorGeo,
              }),
            }),
          });
        } else if(StyleObj.logo == 'diamond') {
          _style = new Style({
            image: new RegularShape({
              radius: Size, // 图形大小
              points: StyleObj.point || 4,
              stroke: new Stroke({
                color: lineColor,
                size: lineSize
              }),
              fill: new Fill({
                color: colorGeo,
              }),
            })
          });
        } else if(StyleObj.logo == 'triangle') {
          _style = new Style({
            image: new RegularShape({ 
              points: StyleObj.point || 3,
              radius: Size, 
              stroke: new Stroke({
                color: lineColor,
                size: lineSize
              }),
              fill: new Fill({
                color: colorGeo,
              }),
            }),
          });
        }
      } else if (StyleObj.shape == 'sCircle') {
        _style = new Style({
          image: new sCircle({ // 动态
            radius: Size, // 动态
            stroke: new Stroke({
              color: lineColor,
              size: lineSize
            }),
            fill: new Fill({
              color: colorGeo,
            }),
          }),
        });
      }

      return _style
    } 
  }
}
